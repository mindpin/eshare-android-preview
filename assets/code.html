require 'rubygems'

class Set
  attr_accessor :id, :name

  def initialize(id, attrs)
    @id = id
    @name = attrs['name']
    @children = []
    @deep = 123
    @reg = /[1-9]\d{5}(?!\d)/
  end

  def build(builder, &block)
    builder.set(:id => @id, :name => @name, :icon => @id, &block)
  end
end
